var config = {
    params:
    {
        logoimage: "Logo.png",
        enableDebugging: "0",
        disableContextMenu: true//屏蔽播放器右键菜单
    }

};
var u = new UnityObject2(config);

//给Unity3D发送消息
function SendMessage(EntityName, FunctionName, Message) {
    u.getUnity().SendMessage(EntityName, FunctionName, Message);
}
//接收Unity3D发送的消息（此函数给Unity3D调用）
function ReadMessage(Message) {
    window.external.ReadMessages(Message);
}
